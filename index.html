from lian import LocalLianClient

mem = LocalLianClient()

mem.remember(
  agent_id="analyst-1",
  content="NVDA guidance raised to $40B",
  event_time=earnings_date
)

mem.recall_at(
  agent_id="analyst-1",
  query="NVDA revenue guidance",
  as_of=march_1_2025
)
